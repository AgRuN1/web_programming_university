Базовые команды

mov перемещает данные
add складывает два значения и помещает в первый регистр
sub вычитает второе значение из первого и сохраняет результат в первый регистр

mov rax,rbx

mov rax,[addr] //значение по адресу

mov [addr],rax

mov rax,5

нельзя перемещать данные из адреса в адрес

sub rsp,28

add rsp,28
